\begin{problem}
    Show that $\{a_n\}_{n\in\nats} \subseteq A$ converges to $a \in A \iff$ for all $U \subseteq A$ which are open in $A$ and contain $a$, there exists some $N \in \nats$ such that $a_n \in U$ for all $n \geq N$.
\end{problem} \vspace{10pt}

\begin{problem}
    If $f: X \to Y$ is a homeomorphism and $A \subseteq X$, then $f: A \to f(A)$ is a homeomorphism.
\end{problem} \vspace{10pt}

\begin{problem}
    $f: \reals^n \to \reals^m$ is continuous $\iff$ $f \circ \iota_A$ is continuous for all $\varnothing \ne A \subsetneq X$.
\end{problem} \vspace{10pt}

\begin{problem}
    If $f: A \to B$ is continuous and $B \subseteq \reals^m$, then $f: A \to \reals^m$ is continuous.
\end{problem} \vspace{10pt}

\begin{problem}
    Show that if $A \subseteq \reals^n$ is sequentially compact and $A \cong B$ then $B$ is sequentially compact. Deduce that $[0, 1]$ is not homeomorphic to $\reals$ or $(0, 1]$.
\end{problem} \vspace{10pt}

\begin{problem}
    Suppose that $\{a_n\}_{n\in\nats} \subseteq A$ converges to $a \in A$, where $A$ is discrete. What can you say about the $a_n$? What can you say about Cauchy sequences in $A$?
\end{problem} \vspace{10pt}

\begin{problem}
    Prove the dyadic rationals $\ints[2^{-1}] = \{ n2^{-m} \ | \ n \in \ints \ m \in \nats \}$ are dense.
\end{problem} \vspace{10pt}

\begin{problem}
    Suppose that $f: X \to Y$ is a homeomorphism. Show that the restriction of $f$ to $X \setminus \{x\}$ is a homeomorphism onto its image for any $x \in X$.
\end{problem} \vspace{10pt}

\begin{problem}
    Show that, \begin{enumerate}[label=(\alph*)]
        \item $\reals \cong \Delta$, where $\Delta = \{ (x, x) \ | \ x \in \reals \}$.
        \item $\reals \cong \reals^+$.
        \item $A \cong B$ if $A, B$ are discrete and $|A| = |B|$.
    \end{enumerate}
\end{problem}
