\begin{problem}
    Suppose $X, Y \subseteq \reals^n$ are connected. Is $\varnothing \ne X \cap Y$ also connected?
\end{problem}

\begin{problem}
    Suppose $U \subseteq \reals^n$ is open and connected. Prove it is also path-connected.
\end{problem}

\begin{problem}
    Let $U \subseteq \reals$ be open and $f: U \to \reals$ be a differentiable function such that $f' \equiv 0$. Show that $f$ is locally constant. Does this mean $f$ is constant?
\end{problem}

\begin{problem}
    A subset $X \subseteq \reals^n$ is called extremally disconnected if $\overline{U}$ is open in $X$ whenever $U$ is open in $X$. Then, \begin{enumerate}
        \item Show that every extremally disconnected is totally disconnected.
        \item Show that $\q$ and $\{ 0 \} \cup \{ 1/n \ | \ n \in \ints^+\}$ are not extremely disconnected.
        \item Show that every discrete space is also extremally disconnected.
        \item Show the converse of (3). \textit{Hint:} Use (2).
    \end{enumerate}
\end{problem}

\begin{problem}
    Prove or disprove: if $X$ is totally disconnected then $X$ is disconnected.
\end{problem}

\begin{problem}
    A topological space is called hyperconnected or anti-Hausdorff, if it cannot be written as the union of any proper open sets. Find all hyperconnected subsets of $\reals^n$. \textit{Hint:} Show that any two open subsets of such a space must intersect non-trivially.
\end{problem}
